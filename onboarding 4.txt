import 'package:flutter/material.dart';

void main() {
  runApp(const DigitalAssetApp());
}

class DigitalAssetApp extends StatelessWidget {
  const DigitalAssetApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Digital Assets',
      debugShowCheckedModeBanner: false,
      theme: ThemeData.dark(),
      home: const MainView(),
    );
  }
}

class MainView extends StatelessWidget {
  const MainView({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: const Color(0xFF0A0A0A),
      body: SafeArea(
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.stretch,
          children: const [
            Expanded(flex: 5, child: AssetVisualization()),
            Divider(thickness: 1, color: Color(0xFF2A2A2A)),
            Expanded(flex: 2, child: DescriptionPanel()),
            Divider(thickness: 1, color: Color(0xFF2A2A2A)),
            Expanded(flex: 1, child: NavigationControls()),
          ],
        ),
      ),
    );
  }
}

class AssetVisualization extends StatelessWidget {
  const AssetVisualization({super.key});

  @override
  Widget build(BuildContext context) {
    return Container(
      color: const Color(0xFF121212),
      padding: const EdgeInsets.all(20),
      child: Center(
        child: Image.network(
          'https://raw.githubusercontent.com/OmarSoaida/images/main/News%20Images%20(2).png',
          fit: BoxFit.contain,
        ),
      ),
    );
  }
}

class DescriptionPanel extends StatelessWidget {
  const DescriptionPanel({super.key});

  @override
  Widget build(BuildContext context) {
    return const Padding(
      padding: EdgeInsets.symmetric(horizontal: 24, vertical: 16),
      child: Align(
        alignment: Alignment.center,
        child: Text(
          'Experience seamless digital asset management with our intuitive platform designed for modern traders.',
          textAlign: TextAlign.center,
          style: TextStyle(
            color: Color(0xFF9C6CFF),
            fontSize: 15,
            fontWeight: FontWeight.w400,
            fontStyle: FontStyle.normal,
            height: 1.4,
          ),
        ),
      ),
    );
  }
}

class NavigationControls extends StatelessWidget {
  const NavigationControls({super.key});

  @override
  Widget build(BuildContext context) {
    return Padding(
      padding: const EdgeInsets.symmetric(horizontal: 28.0, vertical: 8.0),
      child: Row(
        mainAxisAlignment: MainAxisAlignment.spaceBetween,
        children: [
          _buildProgressIndicators(),
          _buildActionButton(),
        ],
      ),
    );
  }

  Widget _buildProgressIndicators() {
    return Row(
      children: List.generate(3, (index) {
        return Container(
          margin: const EdgeInsets.symmetric(horizontal: 5),
          width: 10,
          height: 10,
          decoration: BoxDecoration(
            shape: BoxShape.circle,
            color: index == 2 ? const Color(0xFF4285F4) : const Color(0xFF555555),
          ),
        );
      }),
    );
  }

  Widget _buildActionButton() {
    return TextButton(
      onPressed: () {},
      style: TextButton.styleFrom(
        backgroundColor: const Color(0xFF4285F4),
        padding: const EdgeInsets.symmetric(horizontal: 30, vertical: 14),
        shape: RoundedRectangleBorder(
          borderRadius: BorderRadius.circular(8),
        ),
      ),
      child: const Text(
        'Begin Journey',
        style: TextStyle(
          color: Colors.white,
          fontSize: 15,
          fontWeight: FontWeight.w600,
        ),
      ),
    );
  }
}
